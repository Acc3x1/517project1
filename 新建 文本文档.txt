```java
package dao;

import po.Teacher;
import java.util.List;

public interface TeacherDao {
    List<Teacher> getAllTeacher();
    int addTeacher(Teacher teacher);
    int delTeacherByTno(String tno);
    int updateTeacher(Teacher teacher);
```

### 2. TeacherDaoImpl.java（DAO 实现）

```java
package dao;

import po.Teacher;
import util.DBUtil;
import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.util.ArrayList;
import java.util.List;

public class TeacherDaoImpl implements TeacherDao {

    @Override
    public List<Teacher> getAllTeacher() {
        List<Teacher> list = new ArrayList<>();
        Connection con = DBUtil.getConnection();
        String sql = "SELECT * FROM Teacher";
        try {
            PreparedStatement pst = con.prepareStatement(sql);
            ResultSet rs = pst.executeQuery();
            while (rs.next()) {
                Teacher t = new Teacher();
                t.setTno(rs.getString("tno"));
                t.setTname(rs.getString("tname"));
                t.setAge(rs.getInt("age"));
                t.setSex(rs.getString("sex"));
                t.setPlace(rs.getString("place"));
                list.add(t);
            }
            DBUtil.close(rs, pst);
        } catch (Exception e) {
            e.printStackTrace();
        }
        return list;
    }

    @Override
    public int addTeacher(Teacher teacher) {
        Connection con = DBUtil.getConnection();
        String sql = "INSERT INTO Teacher(tno, tname, age, sex, place) VALUES(?,?,?,?,?)";
        try {
            PreparedStatement pst = con.prepareStatement(sql);
            pst.setString(1, teacher.getTno());
            pst.setString(2, teacher.getTname());
            pst.setInt(3, teacher.getAge());
            pst.setString(4, teacher.getSex());
            pst.setString(5, teacher.getPlace());
            int result = pst.executeUpdate();
            DBUtil.close(pst);
            return result;
        } catch (Exception e) {
            e.printStackTrace();
        }
        return 0;
    }

    @Override
    public int delTeacherByTno(String tno) {
        Connection con = DBUtil.getConnection();
        String sql = "DELETE FROM Teacher WHERE tno=?";
        try {
            PreparedStatement pst = con.prepareStatement(sql);
            pst.setString(1, tno);
            int result = pst.executeUpdate();
            DBUtil.close(pst);
            return result;
        } catch (Exception e) {
            e.printStackTrace();
        }
        return 0;
    }

    @Override
    public int updateTeacher(Teacher teacher) {
        Connection con = DBUtil.getConnection();
        String sql = "UPDATE Teacher SET tname=?, age=?, sex=?, place=? WHERE tno=?";
        try {
            PreparedStatement pst = con.prepareStatement(sql);
            pst.setString(1, teacher.getTname());
            pst.setInt(2, teacher.getAge());
            pst.setString(3, teacher.getSex());
            pst.setString(4, teacher.getPlace());
            pst.setString(5, teacher.getTno());
            int result = pst.executeUpdate();
            DBUtil.close(pst);
            return result;
        } catch (Exception e) {
            e.printStackTrace();
        }
        return 0;
    }
}
```

### 4. TeacherServiceImpl.java（Service 实现）

```java
package service;

import dao.TeacherDao;
import dao.TeacherDaoImpl;
import po.Teacher;
import java.util.List;

public class TeacherServiceImpl implements TeacherService {

    private TeacherDao dao = new TeacherDaoImpl();

    @Override
    public List<Teacher> getAllTeacher() {
        return dao.getAllTeacher();
    }

    @Override
    public int addTeacher(Teacher teacher) {
        return dao.addTeacher(teacher);
    }

    @Override
    public int delTeacherByTno(String tno) {
        return dao.delTeacherByTno(tno);
    }

    @Override
    public int updateTeacher(Teacher teacher) {
        return dao.updateTeacher(teacher);
    }
}
```

### 5. TeacherController.java（Controller）

```java
package controller;

import po.Teacher;
import service.TeacherService;
import service.TeacherServiceImpl;

import javax.servlet.http.HttpServletRequest;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

public class TeacherController {

    private TeacherService service = new TeacherServiceImpl();

    public Map<String, Object> getAllTeacher(HttpServletRequest request) {
        List<Teacher> list = service.getAllTeacher();
        Map<String, Object> result = new HashMap<>();
        result.put("total", list.size());
        result.put("rows", list);
        return result;
    }

    public String AddTeacher(HttpServletRequest request) {
        Teacher t = new Teacher();
        t.setTno(request.getParameter("tno"));
        t.setTname(request.getParameter("tname"));
        t.setAge(Integer.parseInt(request.getParameter("age")));
        t.setSex(request.getParameter("sex"));
        t.setPlace(request.getParameter("place"));
        int r = service.addTeacher(t);
        return r > 0 ? "添加成功" : "添加失败";
    }

    public String delTeacherByTno(HttpServletRequest request) {
        String tno = request.getParameter("tno");
        int r = service.delTeacherByTno(tno);
        return r > 0 ? "删除成功" : "删除失败";
    }

    public String updateTeacher(HttpServletRequest request) {
        Teacher t = new Teacher();
        t.setTno(request.getParameter("tno"));
        t.setTname(request.getParameter("tname"));
        t.setAge(Integer.parseInt(request.getParameter("age")));
        t.setSex(request.getParameter("sex"));
        t.setPlace(request.getParameter("place"));
        int r = service.updateTeacher(t);
        return r > 0 ? "修改成功" : "修改失败";
    }
}
```
